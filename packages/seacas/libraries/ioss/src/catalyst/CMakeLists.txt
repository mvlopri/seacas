SET(HEADERS "")
SET(SOURCES "")

TRIBITS_SET_AND_INC_DIRS(DIR ${CMAKE_CURRENT_SOURCE_DIR})
APPEND_GLOB(HEADERS ${DIR}/*.h*)
APPEND_GLOB(SOURCES ${DIR}/*.C)
APPEND_GLOB(HEADERS ${DIR}/blockmesh/*.h*)
APPEND_GLOB(SOURCES ${DIR}/blockmesh/*.C)

TRIBITS_INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}/../"
  "${CMAKE_CURRENT_BINARY_DIR}/../"
)

list(APPEND HEADERS
  "${CMAKE_CURRENT_BINARY_DIR}/iocatalyst_export.h")

TRIBITS_ADD_LIBRARY(
	Iocatalyst
	HEADERS	${HEADERS}
	SOURCES ${SOURCES}
	DEPLIBS Ioss ${DEPLIBS}
)

IF (BUILD_TESTING)
  ENABLE_TESTING()
ENDIF (BUILD_TESTING)
